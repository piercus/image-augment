Examples
// Simple usage, pad by 10px, on every side
ia.pad(10);
// pad by 10%, with transparent background
ia.pad({
	percent: 0.1,
	borderType: "transparent"
});
// pad by 10% (left, right), 20% (top, bottom), with constant blue background
ia.pad({
	percent: [0.1, 0.2],
	borderType: "transparent",
	borderValue: [0, 0, 255]
});
